import matplotlib.pyplot as plt
import numpy as np

# Crear datos ficticios
x = np.arange(1, 11)
y1 = np.array([0, 5, 10, 5, 10, 15, 10, 5, 0, 0])
y2 = np.array([0, 0, 0, 5, 10, 5, 10, 15, 10, 5])
y3 = np.array([5, 10, 15, 10, 5, 0, 0, 0, 0, 0])

# Crear un gr√°fico de flujo
fig, ax = plt.subplots()
ax.fill_between(x, 0, y1, label='Categor√≠a', color='blue', alpha=0.7)
ax.fill_between(x, y1, y1 + y2, label='Categor√≠a 2', color='green', alpha=0.7)
ax.fill_between(x, y1 + y2, y1 + y2 + y3, label='Categor√≠a 3', color='orange', alpha=0.7)

# Personalizar el gr√°fico
ax.set_xlabel('Tiempo')
ax.set_ylabel('Valor')
ax.set_title('Gr√°fico de Flujo (Streamgraph)')
ax.legend(loc='upper right')

# Mostrar el gr√°fico
plt.show()
